.EQU PORTD =$12      ; PORT D
.EQU PINB =$16       ; PORT B
.EQU DDRB =$17       ; PORT B SMER
.EQU PORTB =$18      ; PORT B VYSTUP - LEDky
.EQU SPL = $3D  ; Stackpointer lowbyte
.EQU SPH = $3E ; Stackpointer highbyte

; Definice promenych  (volne registry 16-31 - viz dataheet str 13.)
